//Combine runestone with weapon or handwraps
//Checks if runestone is being combined
IF
RequestCanCombine((CHARACTER)_Character, (ITEM)_Runestone, (ITEM)_Object, _, _, _, (INTEGER)_RequestID)
AND
DB_RunesAll((STRING)_Status, (ITEMROOT)_RunestoneTemplate)
AND
TemplateIsInPartyInventory((ITEMROOT)TOOL_GEN_RunesmithingTools_14053a44-611e-4b7d-a853-9beac0aa6c1a, _Character, _Count, 0)
AND
_Count > 0
AND
GetTemplate(_Runestone, _Template)
AND
_Template == _RunestoneTemplate
AND
HasActiveStatus(_Object, _Status, 0)
THEN
ApplyStatus(_Object, _Status, -1.0);

//Remove runestone when rune is applied
IF
StatusApplied((ITEM)_Object, (STRING)_Status, _, _)
AND
DB_RunesAll(_Status, (ITEMROOT)_Runestone)
AND
GetOwner(_Object, _Owner)
THEN
MagicPocketsDestroyLocalItemsByTemplate(_Owner, _Runestone, 1);
NOT DB_ApplyRune(_Object, _Status, _Runestone);
DB_ApplyRune(_Object, _Status, _Runestone);
TimerLaunch("RuneAppliedTimer", 1000);

//When weapon potency is applied, also apply PROPERTY_RUNES_0
IF
StatusApplied((ITEM)_Object, (STRING)_Status, _, _)
AND
DB_RunesAll(_Status, (ITEMROOT)_Runestone)
AND
IsSubstring(_Status, "WEAPON_POTENCY", 1)
THEN
ApplyStatus(_Object, "PROPERTY_RUNES_0", -1.0);

//When property runes are applied, increase PROPERTY_RUNES_X by 1
IF
TimerFinished("RuneAppliedTimer")
THEN
PROC_PropertyIncrease_2();
PROC_PropertyIncrease_1();

PROC
PROC_PropertyIncrease_1()
AND
DB_ApplyRune(_Object, _Status, _Runestone)
AND
NOT DB_RunesFundamental(_Status)
AND
HasActiveStatus(_Object, "PROPERTY_RUNES_0", 1)
THEN
ApplyStatus(_Object, "PROPERTY_RUNES_1", -1.0);
NOT DB_ApplyRune(_Object, _Status, _Runestone);

PROC
PROC_PropertyIncrease_2()
AND
DB_ApplyRune(_Object, _Status, _Runestone)
AND
NOT DB_RunesFundamental(_Status)
AND
HasActiveStatus(_Object, "PROPERTY_RUNES_1", 1)
THEN
ApplyStatus(_Object, "PROPERTY_RUNES_2", -1.0);
NOT DB_ApplyRune(_Object, _Status, _Runestone);









//Remove all shield and property runes by combining with runesmithing tools
IF
RequestCanCombine((CHARACTER)_Character, (ITEM)_Tools, (ITEM)_Object, _, _, _, (INTEGER)_RequestID)
AND
TemplateIsInPartyInventory((ITEMROOT)TOOL_GEN_RunesmithingTools_14053a44-611e-4b7d-a853-9beac0aa6c1a, _Character, _Count, 0)
AND
_Count > 0
AND
GetItemByTemplateInPartyInventory((ITEMROOT)TOOL_GEN_RunesmithingTools_14053a44-611e-4b7d-a853-9beac0aa6c1a, _Character, (ITEM)_ToolsTemplate)
AND
_Tools == _ToolsTemplate
AND
DB_RunesAll((STRING)_Status, _)
AND
_Status != "WEAPON_POTENCY"
AND
_Status != "WEAPON_POTENCY_2"
AND
_Status != "STRIKING"
AND
_Status != "STRIKING_GREATER"
AND
HasActiveStatus(_Object, _Status, 1)
THEN
RemoveStatus(_Object, _Status);
PROC_ResetPropertyStatus(_Object);

PROC
PROC_ResetPropertyStatus((ITEM)_Object)
AND
IsTagged(_Object, (TAG)WPN_SHIELD_dc9e09f0-b2ba-4720-bd28-ee9ddbdef2f1, 0)
THEN
RemoveStatus(_Object, "PROPERTY_RUNES_2");
RemoveStatus(_Object, "PROPERTY_RUNES_1");
ApplyStatus(_Object, "PROPERTY_RUNES_0", -1.0);

//Return runestone when rune is removed or upgraded to a higher version
IF
StatusRemoved((ITEM)_Object, (STRING)_Status, _, _)
AND
DB_RunesAll(_Status, (ITEMROOT)_Runestone)
AND
GetOwner(_Object, _Owner)
THEN
TemplateAddTo(_Runestone, _Owner, 1);
